sub opredelenie() {
	($rod,$chlen) = @_; # $rod 1 - zhenski, 2 - sreden, $chlen 1 - kratyk
	$x = $data_opredelenia[rand($#data_opredelenia)];
	if ($rod == 0) {
		$x .= 'ата';
	} elsif ($rod == 1) {
		$x .= 'ото';
	} elsif ($rod == 2) {
		$x .= 'ия';
	} elsif ($chlen == 1) {
		$x .= 'ия';
	} else {
		$x .= 'ият';
	}
}

sub skazuemo() {
	return $data_skazuemo[rand($#data_skazuemo)];
}

sub sashtestvitelno() {
	return $data_sashtestvitelno[rand($#data_sashtestvitelno)];
}

sub podlog() {
	$x = &sashtestvitelno;
	$last = substr($x,-1,1);
	if ($last eq 'а') {
		$x = &opredelenie(1,0);
	} elsif ($last eq 'е') {
		$x = &opredelenie(2,0);
	} else {
		$x = &opredelenie(3,1);
	}
	return $x;
}

sub dopalnenie() {
	if (rand(10) % 2 == 0) {
		return $data_dopalnenie[rand($#data_dopalnenie)];
	} else {
		&podlog . ' ' . &sashtestvitelno;
	}
}

sub izrechenie() {
	$x = &podlog.' '.&skazuemo.' '.&dopalnenie;
	return $x;
}

print "Content-type: text/plain; charset=UTF-8\n\n";

for ($i=0;$i<40;$i++) {
	print &izrechenie;
	print "\n";
